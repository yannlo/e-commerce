    TypeError:

        code: 3xx -> object or element type is invalid
        code: 4xx -> not confirm action
        code: 5xx -> not found or not exist the object or the element
        code: 6xx -> impossible action
        code: 7xx -> failed action
        code: 8xx -> inconsistent result
